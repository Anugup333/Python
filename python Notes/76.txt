#Program for Accepting a Value from KBD and Decide whether It is Palindrome or not
#TernaryOpEx1.py
val=input("Enter a Value:") # val=LIRIL
res =  "Palindrome" if  val==val[::-1] else  "Not Palindrome"
print("{} is {}".format(val,res))

#Program for accepting Two Numerical Values and find Biggest among them
# and check for equality

#Program for accepting Three Numerical Values and find Biggest among them
# and check for equality
#TernaryOpEx4.py
a=float(input("Enter Value of a:"))
b=float(input("Enter Value of b:"))
c=float(input("Enter Value of c:"))
#Find Biggest among them and check for equality
res=a if (a>=b and a>c) else b if (b>a and b>=c) else c if (c>=a and c>b) else "All Values Equal"
print("big({},{},{})={}".format(a,b,c,res))
